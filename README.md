# streamwrapped

pip install flask
pip install python-steam-api
pip install python-dotenv
pip install pymongo

first terminal:
cd backend/src

python infrastructure.py

second terminal:
cd backend

npm install nodemon
npm install cors
npm install mongodb
npm install dotenv

npm run dev

third terminal:
cd frontend/client
npm install next@latest react@latest react-dom@latest @react-oauth/google
npm install jwt-decode
npm install axios
npm run dev

add .env file with STEAM_API_KEY
